#include <iostream>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;

    int a[50];

    for (int i = 0; i < n; i++)
        cin >> a[i];

    int counter = 0;

    for (int i = 0; i < n; i++) {
        if (a[i] >= a[k - 1] && a[i] > 0)
            counter++;
    }

    cout << counter << endl;
}
