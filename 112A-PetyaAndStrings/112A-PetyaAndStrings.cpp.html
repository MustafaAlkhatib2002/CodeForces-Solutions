#include <iostream>
using namespace std;
#include <cmath>
#include <ctype.h>
#include <string>
 
int main() {
 
    string st1, st2;
    cin >> st1 >> st2;
 
 
    for (int i = 0; i < st1.length(); i++) {
 
        st1.at(i) = tolower(st1.at(i));
        st2.at(i) = tolower(st2.at(i));
 
    }
 
 
    for (int i = 0; i < st1.length(); i++) {
 
        if (st1.at(i) > st2.at(i)) {
            cout<<1;
            return 0;
        }
        else if (st2.at(i) > st1.at(i)) {
            cout<<-1;
            return 0;
        }
 
    }
 
 
    cout<<0;
 
    return 0;
 
}
